\begin{frame}{价格型因子}
\begin{itemize}
  \item 自回归因子$~A_{t-k}~$
  \item 均线因子 $~MA(t)=\frac{1}{K}\sum^K_{i=1}A(t-i)~$
  \item 价格分位数 $~Percentile(A_t)~$
  \item 现货价格 $~A^*_{t}~$
  \item 交易量分位数 $~Percentile(V_t)~$
\end{itemize}
\end{frame}


\begin{frame}{时间型因子}
\begin{itemize}
  \item 交割时间
  \item 交割时间差
\end{itemize}
\end{frame}


\begin{frame}{波动率型因子}
\begin{itemize}
  \item 极差 $~\max{A_{t_i}}-\min{A_{t_j}}~$
  \item 收益波动率 $~\bar{\Delta{A_t}}~$
\end{itemize}
\end{frame}
